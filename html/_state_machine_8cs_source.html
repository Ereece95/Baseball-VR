79  public StateMachine(StateMachineRunner engine, MonoBehaviour component)
80  {
81  this.engine = engine;
82  this.component = component;
83 
84  //Define States
85  var values = Enum.GetValues(typeof(T));
86  if (values.Length < 1) { throw new ArgumentException("Enum provided to Initialize must have at least 1 visible definition"); }
87 
88  stateLookup = new Dictionary<object, StateMapping>();
89  for (int i = 0; i < values.Length; i++)
90  {
91  var mapping = new StateMapping((Enum) values.GetValue(i));
92  stateLookup.Add(mapping.state, mapping);
93  }
94 
95  //Reflect methods
96  var methods = component.GetType().GetMethods(BindingFlags.Instance | BindingFlags.DeclaredOnly | BindingFlags.Public |
97  BindingFlags.NonPublic);
98 
99  //Bind methods to states
100  var separator = "_".ToCharArray();
101  for (int i = 0; i < methods.Length; i++)
102  {
103  if (methods[i].GetCustomAttributes(typeof(CompilerGeneratedAttribute), true).Length != 0)
104  {
105  continue;
106  }
107 
108  var names = methods[i].Name.Split(separator);
109 
110  //Ignore functions without an underscore
111  if (names.Length <= 1)
112  {
113  continue;
114  }
115 
116  Enum key;
117  try
118  {
119  key = (Enum) Enum.Parse(typeof(T), names[0]);
120  }
121  catch (ArgumentException)
122  {
123  //Not an method as listed in the state enum
124  continue;
125  }
126 
127  var targetState = stateLookup[key];
128 
129  switch (names[1])
130  {
131  case "Enter":
132  if (methods[i].ReturnType == typeof(IEnumerator))
133  {
134  targetState.hasEnterRoutine = true;
135  targetState.EnterRoutine = CreateDelegate<Func<IEnumerator>>(methods[i], component);
136  }
137  else
138  {
139  targetState.hasEnterRoutine = false;
140  targetState.EnterCall = CreateDelegate<Action>(methods[i], component);
141  }
142  break;
143  case "Exit":
144  if (methods[i].ReturnType == typeof(IEnumerator))
145  {
146  targetState.hasExitRoutine = true;
147  targetState.ExitRoutine = CreateDelegate<Func<IEnumerator>>(methods[i], component);
148  }
149  else
150  {
151  targetState.hasExitRoutine = false;
152  targetState.ExitCall = CreateDelegate<Action>(methods[i], component);
153  }
154  break;
155  case "Finally":
156  targetState.Finally = CreateDelegate<Action>(methods[i], component);
157  break;
158  case "Update":
159  targetState.Update = CreateDelegate<Action>(methods[i], component);
160  break;
161  case "LateUpdate":
162  targetState.LateUpdate = CreateDelegate<Action>(methods[i], component);
163  break;
164  case "FixedUpdate":
165  targetState.FixedUpdate = CreateDelegate<Action>(methods[i], component);
166  break;
167  case "OnCollisionEnter":
168  targetState.OnCollisionEnter = CreateDelegate<Action<Collision>>(methods[i], component);
169  break;
170  }
171  }
172 
173  //Create nil state mapping
174  currentState = new StateMapping(null);
175  }

207  public void ChangeState(T newState, StateTransition transition)
208  {
209  if (stateLookup == null)
210  {
211  throw new Exception("States have not been configured, please call initialized before trying to set state");
212  }
213 
214  if (!stateLookup.ContainsKey(newState))
215  {
216  throw new Exception("No state with the name " + newState.ToString() + " can be found. Please make sure you are called the correct type the statemachine was initialized with");
217  }
218 
219  var nextState = stateLookup[newState];
220 
221  if (currentState == nextState) return;
222 
223  //Cancel any queued changes.
224  if (queuedChange != null)
225  {
226  engine.StopCoroutine(queuedChange);
227  queuedChange = null;
228  }
229 
230  switch (transition)
231  {
232  //case StateMachineTransition.Blend:
233  //Do nothing - allows the state transitions to overlap each other. This is a dumb idea, as previous state might trigger new changes.
234  //A better way would be to start the two couroutines at the same time. IE don't wait for exit before starting start.
235  //How does this work in terms of overwrite?
236  //Is there a way to make this safe, I don't think so?
237  //break;
238  case StateTransition.Safe:
239  if (isInTransition)
240  {
241  if (exitRoutine != null) //We are already exiting current state on our way to our previous target state
242  {
243  //Overwrite with our new target
244  destinationState = nextState;
245  return;
246  }
247 
248  if (enterRoutine != null) //We are already entering our previous target state. Need to wait for that to finish and call the exit routine.
249  {
250  //Damn, I need to test this hard
251  queuedChange = WaitForPreviousTransition(nextState);
252  engine.StartCoroutine(queuedChange);
253  return;
254  }
255  }
256  break;
257  case StateTransition.Overwrite:
258  if (currentTransition != null)
259  {
260  engine.StopCoroutine(currentTransition);
261  }
262  if (exitRoutine != null)
263  {
264  engine.StopCoroutine(exitRoutine);
265  }
266  if (enterRoutine != null)
267  {
268  engine.StopCoroutine(enterRoutine);
269  }
270 
271  //Note: if we are currently in an EnterRoutine and Exit is also a routine, this will be skipped in ChangeToNewStateRoutine()
272  break;
273  }
274 
275 
276  if ((currentState != null && currentState.hasExitRoutine) || nextState.hasEnterRoutine)
277  {
278  isInTransition = true;
279  currentTransition = ChangeToNewStateRoutine(nextState, transition);
280  engine.StartCoroutine(currentTransition);
281  }
282  else //Same frame transition, no coroutines are present
283  {
284  if (currentState != null)
285  {
286  currentState.ExitCall();
287  currentState.Finally();
288  }
289 
290  lastState = currentState;
291  currentState = nextState;
292  if (currentState != null)
293  {
294  currentState.EnterCall();
295  if (Changed != null)
296  {
297  Changed((T) currentState.state);
298  }
299  }
300  isInTransition = false;
301  }
302  }
309  private IEnumerator ChangeToNewStateRoutine(StateMapping newState, StateTransition transition)
310  {
311  destinationState = newState; //Chache this so that we can overwrite it and hijack a transition
312 
313  if (currentState != null)
314  {
315  if (currentState.hasExitRoutine)
316  {
317  exitRoutine = currentState.ExitRoutine();
318 
319  if (exitRoutine != null && transition != StateTransition.Overwrite) //Don't wait for exit if we are overwriting
320  {
321  yield return engine.StartCoroutine(exitRoutine);
322  }
323 
324  exitRoutine = null;
325  }
326  else
327  {
328  currentState.ExitCall();
329  }
330 
331  currentState.Finally();
332  }
333 
334  lastState = currentState;
335  currentState = destinationState;
336 
337  if (currentState != null)
338  {
339  if (currentState.hasEnterRoutine)
340  {
341  enterRoutine = currentState.EnterRoutine();
342 
343  if (enterRoutine != null)
344  {
345  yield return engine.StartCoroutine(enterRoutine);
346  }
347 
348  enterRoutine = null;
349  }
350  else
351  {
352  currentState.EnterCall();
353  }
354 
355  //Broadcast change only after enter transition has begun.
356  if (Changed != null)
357  {
358  Changed((T) currentState.state);
359  }
360  }
361 
362  isInTransition = false;
363  }
